Calculo de un número aleatorio

	    
	    	//HTML
	      <div class="w-75 text-center">

					   <div>
						      <button type="button" id="btn-seleccionar">Seleccionar un número entre Cero y 100</button>			
					   </div>

					   <div class="my-4">
						      <output id="salida"></output>			
					   </div>

				</div>

					//JS
					<script>
						   const boton = document.getElementById('btn-seleccionar')
						   const salida = document.getElementById('salida')

						   boton.addEventListener('click', () => {

							      let aleatorio = Math.random()*100
							      let entero = Math.floor(aleatorio)

							      salida.style.fontSize = '32px'
							      salida.value = entero							

						   })
					</script>
